			scope:recipient.is_imprisoned = no

			}
		}

				type = event_generic_good_with_text
				title = sanguine_interaction_notification_title

		}
	}

			add_opinion = { 
				target = scope:actor 
				modifier = unfriendly_opinion 
				opinion = -20 
			}
		}
		if = { 
			limit = { 
				scope:actor = { any_secret = { secret_type = MythB_secret_demon } } 
			} 
			scope:actor = {
				random_secret = { 
					limit = { secret_type = MythB_secret_demon } 
					reveal_to = scope:recipient
				}
			}
		}
		scope:actor = {
			add_opinion = { target = scope:recipient modifier = hurt_opinion opinion = -50 }
			send_interface_message = {
				type = event_generic_bad_with_text
				title = sanguine_decline_notification_title
				desc = sanguine_decline_notification_desc
				right_icon = scope:recipient
			}
		}
	}
